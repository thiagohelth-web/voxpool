async function apurarEnquete(enqueteId, opcaoVencedora) { try { // 1. Busca todas as apostas feitas nesta enquete específica na nuvem const { data: apostas, error: errApostas } = await clientSupabase .from('historico_apostas') .select('*') .eq('enquete_id', enqueteId.toString()); if (errApostas) { alert("Erro ao buscar apostas: " + errApostas.message); return; } // 2. Se houver apostas registradas, faz o processamento financeiro if (apostas && apostas.length > 0) { for (const aposta of apostas) { const valorTotal = parseFloat(aposta.valor_total); let retornoCalculado = 0; // CORREÇÃO AQUI: Verifica corretamente a coluna 'opcao_escolhida' if (aposta.opcao_escolhida === opcaoVencedora.toString()) { // Se o membro ganhou: Devolve o valor apostado + Lucro de 49% (Total: 1.49x) retornoCalculado = valorTotal * 1.49; } else { // Se o membro perdeu: Mitiga o dano e devolve apenas 39% do valor apostado retornoCalculado = valorTotal * 0.39; } // Busca o saldo atual do membro dono desta aposta const { data: m } = await clientSupabase .from('membros') .select('saldo') .eq('id', aposta.membro_id) .maybeSingle(); if (m) { // Soma o retorno calculado diretamente ao saldo atual do membro na nuvem const novoSaldo = parseFloat(m.saldo) + retornoCalculado; // Atualiza o saldo na nuvem await clientSupabase .from('membros') .update({ saldo: novoSaldo }) .eq('id', aposta.membro_id); } } // Limpa o histórico de apostas dessa enquete encerrada para não acumular lixo await clientSupabase.from('historico_apostas').delete().eq('enquete_id', enqueteId.toString()); } // 3. Remove a enquete da lista ativa await clientSupabase.from('enquetes').delete().eq('id', enqueteId); alert("Apuração concluída! Saldos atualizados e lucros distribuídos com sucesso."); carregarAdmin(); } catch (err) { console.error(err); alert("Erro durante a apuração."); } }
