async function carregarAdmin() { try { // 1. LISTAR RECARGAS (Lendo da tabela 'depositos') const { data: recargas } = await clientSupabase.from('depositos').select('*'); const tRecargas = document.getElementById('tabela-recargas'); tRecargas.innerHTML = ""; if (!recargas || recargas.length === 0) { tRecargas.innerHTML = "Nenhum depósito pendente."; } else { recargas.forEach(rec => { if(rec.status !== 'APROVADO') { // Ajustado para usar as colunas exatas do banco: valor_rs e membro_id const val = parseFloat(rec.valor_rs || 0); const idMembro = rec.membro_id || "Sem Identificação"; let cl = ""; let st = ""; if (rec.status === "PENDENTE" || rec.status === "Pendente" || rec.status === "envio_confirmado_usuario") { cl = "class='linha-paga'"; st = `💰 PIX ENVIADO`; } else if (rec.status === "RECUSADO" || rec.status === "pix_nao_recebido") { cl = "class='linha-recusada'"; st = `❌ ERRO NOTIFICADO`; } else { st = `⏳ AGUARDANDO`; } tRecargas.innerHTML += ` ${idMembro}
Ref: ${rec.comprovante || 'Nenhum'} ${val.toFixed(2)} R$ ${val.toFixed(2)} ${st} `; } }); if(tRecargas.innerHTML === "") { tRecargas.innerHTML = "Todos os depósitos foram processados!"; } } // 2. LISTAR PEDIDOS DE SAQUE (Lendo da tabela 'saques') const { data: saques } = await clientSupabase.from('saques').select('*').or('status.eq.PENDENTE,status.eq.Pendente'); const tSaques = document.getElementById('tabela-saques'); tSaques.innerHTML = ""; if(!saques || saques.length === 0) { tSaques.innerHTML = "Nenhum pedido de saque pendente."; } else { saques.forEach(saq => { // Ajustado para usar as colunas exatas do banco: membro_id e moedas_sacar const idMembroSaque = saq.membro_id; const moedasSacar = parseFloat(saq.moedas_sacar || 0); tSaques.innerHTML += ` ${idMembroSaque} ${moedasSacar.toFixed(2)} Moedas ${saq.chave_pix} `; }); } // 3. GERENCIADOR DE MEMBROS COMPLETO const { data: membros } = await clientSupabase.from('membros').select('*'); const tMembros = document.getElementById('tabela-membros-geral'); tMembros.innerHTML = ""; if(membros) { membros.forEach(m => { let statusStyle = m.status === 'bloqueado' ? 'background:var(--accent-red); color:white;' : 'background:var(--accent-green); color:black;'; tMembros.innerHTML += ` ${m.id} ${m.chave_pix || 'Não cadastrada'} ${parseFloat(m.saldo || 0).toFixed(2)} ${m.status ? m.status.toUpperCase() : 'ATIVO'} ${m.status === 'bloqueado' ? `` : `` } `; }); } // 4. LISTAR ENQUETES PARA APURAÇÃO const { data: enquetes } = await clientSupabase.from('enquetes').select('*'); const tabela = document.getElementById('tabela-enquetes'); tabela.innerHTML = ""; if(enquetes && enquetes.length > 0) { enquetes.forEach(enq => { let o1 = "Opção 1", o2 = "Opção 2"; let ex = enq.pergunta; if (enq.categoria === "Mercado Financeiro") { o1 = "ACIMA"; o2 = "ABAIXO"; try { if (enq.pergunta.includes(" vs ")) { ex = enq.pergunta.split(" vs ")[0] + "]"; } } catch(e){} } else { if (enq.pergunta.includes("[") && enq.pergunta.includes("vs")) { try { const partes = enq.pergunta.split(" ["); ex = partes[0]; const opcoesRaw = partes[1].replace("]", "").split(" vs "); o1 = opcoesRaw[0]; o2 = opcoesRaw[1]; } catch(e){} } } tabela.innerHTML += ` ${enq.categoria} ${ex} 1: ${o1}
2: ${o2} `; }); } else { tabela.innerHTML = "Nenhuma previsão ativa no momento."; } } catch(e) { console.error("Erro ao carregar dados do admin:", e); } }
