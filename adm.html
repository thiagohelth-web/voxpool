// --- APURAÇÃO COM LOGICA DE RATEIO DO POTE (POOL) OTIMIZADA --- async function apurarEnquete(enqueteId, opcaoVencedora) { if(!confirm("Tem certeza que deseja encerrar esta enquete com a opção selecionada? O cálculo do rateio será definitivo.")) return; try { // Garante que o ID seja tratado corretamente (ajuste se seu ID for UUID) const idBusca = isNaN(enqueteId) ? enqueteId : parseInt(enqueteId); // 1. Busca todas as apostas da enquete const { data: apostas, error: errApostas } = await clientSupabase .from('historico_apostas') .select('*') .eq('enquete_id', idBusca); if (errApostas) throw errApostas; if (apostas && apostas.length > 0) { let poteTotal = 0; let volumeVencedores = 0; let apostasVencedoras = []; // Calcula volumes apostas.forEach(aposta => { const valor = parseFloat(aposta.valor_total || 0); poteTotal += valor; if (aposta.opcao_escolhida?.toString() === opcaoVencedora.toString()) { volumeVencedores += valor; apostasVencedoras.push(aposta); } }); if (volumeVencedores > 0) { // Otimização: Coleta os saldos de TODOS os vencedores de uma só vez para evitar loops infinitos de SELECT const IDsVencedores = apostasVencedoras.map(v => v.membro_id); const { data: membros, error: errMembros } = await clientSupabase .from('membros') .select('id, saldo') .in('id', IDsVencedores); if (errMembros) throw errMembros; // Cria um mapa de saldos para acesso rápido O(1) const saldoMapa = {}; membros.forEach(m => { saldoMapa[m.id] = parseFloat(m.saldo || 0); }); // Executa as atualizações em paralelo usando Promise.all const promessasPagamento = apostasVencedoras.map(async (vencedor) => { let suaProporcao = parseFloat(vencedor.valor_total) / volumeVencedores; let premioFinal = suaProporcao * poteTotal; let saldoAtual = saldoMapa[vencedor.membro_id]; if (saldoAtual !== undefined) { let novoSaldo = saldoAtual + premioFinal; return clientSupabase .from('membros') .update({ saldo: novoSaldo }) .eq('id', vencedor.membro_id); } }); // Aguarda todos os saldos serem atualizados await Promise.all(promessasPagamento); } else { alert("Aviso: Ninguém acertou a previsão! O valor acumulado ficou para a plataforma."); } // Deleta o histórico de apostas apenas DEPOIS que todos receberam await clientSupabase.from('historico_apostas').delete().eq('enquete_id', idBusca); } // Remove a enquete ativa do painel await clientSupabase.from('enquetes').delete().eq('id', idBusca); alert("Apuração Concluída e prêmio rateado com sucesso!"); carregarAdmin(); } catch (error) { console.error("Erro na apuração:", error); alert("Ocorreu um erro ao processar a apuração. Verifique o console."); } }
