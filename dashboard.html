async function solicitarSaque() { const valorSaque = parseFloat(document.getElementById('input-valor-saque').value); const pix = document.getElementById('input-pix-saque').value; // Forçamos o ID para string para garantir que a busca no Supabase funcione const usuarioId = String(localStorage.getItem('vox_usuario')).trim(); if (!valorSaque || isNaN(valorSaque) || valorSaque <= 0) { alert("Por favor, insira um valor de saque válido."); return; } if (!pix) { alert("Por favor, informe sua chave PIX."); return; } try { // 1. Busca saldo atual const { data: membro, error: fetchError } = await clientSupabase .from('membros') .select('saldo') .eq('id', usuarioId) .maybeSingle(); if (fetchError || !membro) { console.error("Erro busca:", fetchError); alert("Erro: Usuário não encontrado ou sessão expirada."); return; } const saldoAtual = parseFloat(membro.saldo || 0); // 2. Validação if (saldoAtual < valorSaque) { alert("Saldo insuficiente. Você possui: R$ " + saldoAtual.toFixed(2)); return; } // 3. Registrar o pedido const { error: insertError } = await clientSupabase.from('saques').insert([ { membro_id: usuarioId, moedas_sacar: valorSaque, chave_pix: pix, status: 'PENDENTE' } ]); if (insertError) throw insertError; // 4. ATUALIZAR SALDO (Dedução) const novoSaldo = saldoAtual - valorSaque; const { error: updateError } = await clientSupabase .from('membros') .update({ saldo: novoSaldo }) .eq('id', usuarioId); if (updateError) throw updateError; alert("Sucesso! Saque solicitado e saldo atualizado."); location.reload(); } catch (err) { console.error("Erro crítico no saque:", err); alert("Ocorreu um erro. Verifique se o seu ID de membro está correto."); } }
