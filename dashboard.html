async function solicitarSaque() { const valorSaque = parseFloat(document.getElementById('input-valor-saque').value); const pix = document.getElementById('input-pix-saque').value; const usuarioId = String(localStorage.getItem('vox_usuario')).trim(); // Diagnóstico: O que estamos enviando para o banco? console.log("DEBUG - ID do Usuário sendo buscado:", usuarioId); if (!valorSaque || isNaN(valorSaque) || valorSaque <= 0) { alert("Por favor, insira um valor de saque válido."); return; } if (!pix) { alert("Por favor, informe sua chave PIX."); return; } try { const { data: membro, error: fetchError } = await clientSupabase .from('membros') .select('id, saldo') // Incluímos o 'id' aqui para conferência .eq('id', usuarioId) .maybeSingle(); if (fetchError) { console.error("Erro do Supabase:", fetchError); alert("Erro de conexão com o banco."); return; } if (!membro) { console.error("Nenhum membro encontrado com o ID:", usuarioId); alert("Erro: O seu ID de usuário ('" + usuarioId + "') não foi localizado no banco de dados."); return; } const saldoAtual = parseFloat(membro.saldo || 0); if (saldoAtual < valorSaque) { alert("Saldo insuficiente. Você possui: R$ " + saldoAtual.toFixed(2)); return; } // Registrar o pedido const { error: insertError } = await clientSupabase.from('saques').insert([ { membro_id: usuarioId, moedas_sacar: valorSaque, chave_pix: pix, status: 'PENDENTE' } ]); if (insertError) throw insertError; // ATUALIZAR SALDO const novoSaldo = saldoAtual - valorSaque; const { error: updateError } = await clientSupabase .from('membros') .update({ saldo: novoSaldo }) .eq('id', usuarioId); if (updateError) throw updateError; alert("Sucesso! Saque solicitado e saldo atualizado."); location.reload(); } catch (err) { console.error("Erro crítico:", err); alert("Ocorreu um erro. Verifique o console (F12) para detalhes."); } }
