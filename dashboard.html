async function solicitarSaque() { // 1. Coleta de dados const valorSaque = parseFloat(document.getElementById('input-valor-saque').value); const pix = document.getElementById('input-pix-saque').value; const usuarioId = String(localStorage.getItem('vox_usuario')).trim(); // 2. Validações básicas if (!valorSaque || isNaN(valorSaque) || valorSaque <= 0) { alert("Valor inválido."); return; } try { // 3. Busca o membro no banco const { data: membro, error: fetchError } = await clientSupabase .from('membros') .select('id, saldo') .eq('id', usuarioId) .maybeSingle(); if (fetchError || !membro) { alert("Erro: Usuário não localizado."); return; } const saldoAtual = parseFloat(membro.saldo || 0); if (saldoAtual < valorSaque) { alert("Saldo insuficiente."); return; } // 4. Insere o pedido na tabela 'saques' const { error: insertError } = await clientSupabase .from('saques') .insert([{ membro_id: usuarioId, moedas_sacar: valorSaque, chave_pix: pix, status: 'PENDENTE' }]); if (insertError) throw insertError; // 5. Atualiza o saldo do membro const { error: updateError } = await clientSupabase .from('membros') .update({ saldo: (saldoAtual - valorSaque) }) .eq('id', usuarioId); if (updateError) throw updateError; alert("Sucesso! Saque solicitado."); location.reload(); } catch (err) { console.error("Erro na função solicitarSaque:", err); alert("Ocorreu um erro técnico. Verifique o console F12."); } }
