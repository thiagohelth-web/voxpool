// FUNÇÃO DE DEPÓSITO REVISADA COM ALERTAS DE ERRO DETALHADOS DO SUPABASE async function enviarNotificacaoDeposito() { try { const valorRs = document.getElementById('dep-valor').value; const moedas = document.getElementById('dep-moedas').value; const refId = document.getElementById('dep-id').value; if(!valorRs || !moedas || !refId.trim()) { alert("Por favor, preencha todos os campos!"); return; } const payload = { membro_id: usuarioLogado, membro: usuarioLogado, valor: parseFloat(valorRs), valor_rs: parseFloat(valorRs), moedas: parseFloat(moedas), comprovante: refId, detalhes: refId, status: 'PENDENTE' }; // Força a tentativa na tabela 'depositos' let { data, error: err1 } = await clientSupabase.from('depositos').insert([payload]).select(); if (err1) { console.error("Erro na tabela 'depositos':", err1); // Tenta a tabela unificada caso a primeira não exista let { error: err2 } = await clientSupabase.from('solicitacoes_financeiras').insert([{ membro_id: usuarioLogado, tipo: 'DEPOSITO', valor: parseFloat(moedas), detalhes: `R$: ${valorRs} - Ref: ${refId}`, status: 'PENDENTE' }]); if(err2) { alert(`Erro no Supabase!\nCódigo: ${err2.code}\nMensagem: ${err2.message}\nDetalhes: ${err2.details}`); return; } } alert("Notificação enviada com sucesso no código! Verifique se apareceu no ADM."); document.getElementById('dep-valor').value = ""; document.getElementById('dep-moedas').value = ""; document.getElementById('dep-id').value = ""; document.getElementById('painel-deposito').style.display = 'none'; } catch (err) { alert("Erro inesperado no JavaScript: " + err.message); } } // FUNÇÃO DE SAQUE REVISADA COM ALERTAS DE ERRO DETALHADOS DO SUPABASE async function enviarSolicitacaoSaque() { try { const valor = document.getElementById('saq-valor').value; const chavePix = document.getElementById('saq-chave').value; if(!valor || !chavePix.trim()) { alert("Preencha todos os campos!"); return; } const valorSaque = parseFloat(valor); const { data: m, error: errSaldo } = await clientSupabase.from('membros').select('saldo').eq('id', usuarioLogado).maybeSingle(); if(!m || parseFloat(m.saldo) < valorSaque) { alert("Saldo insuficiente ou erro ao verificar saldo atual."); return; } const payloadSaque = { membro_id: usuarioLogado, membro: usuarioLogado, valor: valorSaque, moedas: valorSaque, moedas_sacar: valorSaque, chave_pix: chavePix, status: 'PENDENTE' }; let { error: err1 } = await clientSupabase.from('saques').insert([payloadSaque]); if(err1) { console.error("Erro na tabela 'saques':", err1); let { error: err2 } = await clientSupabase.from('solicitacoes_financeiras').insert([{ membro_id: usuarioLogado, tipo: 'SAQUE', valor: valorSaque, detalhes: `Chave PIX: ${chavePix}`, status: 'PENDENTE' }]); if(err2) { alert(`Erro no Supabase!\nCódigo: ${err2.code}\nMensagem: ${err2.message}`); return; } } alert("Pedido de saque enviado com sucesso no código!"); document.getElementById('saq-valor').value = ""; document.getElementById('saq-chave').value = ""; document.getElementById('painel-saque').style.display = 'none'; carregarPainel(); } catch (err) { alert("Erro inesperado no JavaScript: " + err.message); } }
