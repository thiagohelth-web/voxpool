async function enviarSolicitacaoSaque() { try { const valor = document.getElementById('saq-valor').value; const chavePix = document.getElementById('saq-chave').value; if(!valor || !chavePix.trim()) { alert("Preencha todos os campos do saque!"); return; } const valorSaque = parseFloat(valor); let saldoAtual = 0; // Busca o saldo atual do usuário para validar const { data: m, error: mErr } = await clientSupabase .from('membros') .select('saldo') .eq('id', usuarioLogado) .maybeSingle(); if (mErr) throw mErr; saldoAtual = parseFloat(m.saldo || 0); if(saldoAtual < valorSaque) { alert("Saldo insuficiente! Você possui: R$ " + saldoAtual.toFixed(2)); return; } // 1. Insere o pedido de saque const { error: insertError } = await clientSupabase .from('saques') .insert([{ membro_id: usuarioLogado, moedas_sacar: valorSaque, chave_pix: chavePix, status: 'PENDENTE' }]); if (insertError) throw insertError; // 2. Deduz o saldo do usuário const { error: updateError } = await clientSupabase .from('membros') .update({ saldo: (saldoAtual - valorSaque) }) .eq('id', usuarioLogado); if (updateError) throw updateError; alert("Pedido de saque enviado com sucesso! O saldo foi atualizado."); document.getElementById('saq-valor').value = ""; document.getElementById('saq-chave').value = ""; document.getElementById('painel-saque').style.display = 'none'; // Recarrega o saldo na tela carregarPainel(); } catch (err) { console.error("Erro técnico:", err); alert("Erro ao processar saque: " + err.message); } }
